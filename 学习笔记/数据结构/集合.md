# 集合



### 实现

```js
let SET = (function () {
    let sym = Symbol()
    return class {
        constructor(arr) {
            // 存储数据
            this[sym] = []
            // this.size = 0
            if (Array.isArray(arr)) {
                // 遍历
                arr.forEach(item => {
                    // 去重
                    if (!this[sym].includes(item)) {
                        this[sym].push(item)
                        // this.size++
                    }
                })
            }
        }
        // 添加
        add(data) {
            // 去重
            if (!this[sym].includes(data)) {
                this[sym].push(data)
                // this.size++
            }
        }
        // 删除数据
        delete(data){
            let i = this[sym].indexOf(data)
            if(i < 0) return
            this[sym].splice(i, 1)
        }
        // 检查集合中是否存在该数据
        has(data){
            return this[sym].includes(data)
        }
        // 长度
        size(){
            return this[sym].length
        }
    }
}())
```
