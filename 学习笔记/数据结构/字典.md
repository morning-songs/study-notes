# 字典



### 实现

```js
let MAP = (function(){
    const sym = Symbol()
    return class{
        constructor(){
            this[sym] = []
        }
        // 添加值 / 修改值
        set(key, value){
            for(let i = 0, len = this.size(); i < len; i++){
                let item = this[sym][i]
                let k = item[0]
                if(k === key){
                    // 重复的key
                    item[1] = value
                    return 
                }
            }
            // [key, value]
            this[sym].push([key, value])
        }

        // 取值
        get(key){
            for(let i = 0, len = this.size(); i < len; i++){
                let item = this[sym][i],
                    k = item[0],
                    v = item[1];
                if(key === k) return v
            }
        }

        // 判断是否存在
        has(key){
            for(let i = 0, len = this.size(); i < len; i++){
                let k = this[sym][i][0]
                if(k === key) return true
            }
            return false
        }

        delete(key){
            for(let i = 0, len = this.size(); i < len; i++){
                let k = this[sym][i][0]
                if(k === key) {
                    this[sym].splice(i, 1)
                    return
                }
            }
        }

        // 获取所有的键
        keys(){
            let arr = []
            this[sym].forEach(([key]) => {
                arr.push(key)
            })
            return arr
        }

        // 获取所有的值
        values(){
            let arr = []
            this[sym].forEach(([, value]) => {
                arr.push(value)
            })
            return arr
        }

        size(){
            return this[sym].length
        }
    }
}())
```
